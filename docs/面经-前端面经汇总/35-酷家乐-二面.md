笔试题

  1. 
```js
    const a = { b: 3 }
    
    function foo(obj) {
      obj.b = 5
    
      return obj
    }
    
    const aa = foo(a)
    
    console.log(a.b)
    
    console.log(aa.b)
```

  2. 
```js
    function Ofo() {}
    
    function Bick() {
      this.name = 'mybick'
    }
    
    var myBick = new Ofo()
    
    Ofo.prototype = new Bick()
    
    var youbick = new Bick()
    
    console.log(myBick.name)
    
    console.log(youbick.name)
```
