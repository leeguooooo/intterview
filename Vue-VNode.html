еҰӮдёҠпјҢз”ЁжқҘжҸҸиҝ°з»„д»¶зҡ„ VNode е…¶ tag еұһжҖ§еҖјеј•з”Ёзҡ„е°ұжҳҜз»„д»¶зұ»(жҲ–еҮҪж•°)жң¬иә«пјҢиҖҢдёҚжҳҜж ҮзӯҫеҗҚз§°еӯ—з¬ҰдёІгҖӮжүҖд»ҘзҗҶи®әдёҠпјҡжҲ‘д»¬еҸҜд»ҘйҖҡиҝҮжЈҖжҹҘtag еұһжҖ§еҖјжҳҜеҗҰжҳҜеӯ—з¬ҰдёІжқҘзЎ®е®ҡдёҖдёӘ VNode жҳҜеҗҰжҳҜжҷ®йҖҡж ҮзӯҫгҖӮ

йҷӨдәҶз»„д»¶д№ӢеӨ–пјҢиҝҳжңүдёӨз§ҚжҠҪиұЎзҡ„еҶ…е®№йңҖиҰҒжҸҸиҝ°пјҢеҚі Fragment е’Ң PortalгҖӮжҲ‘д»¬е…ҲжқҘдәҶи§ЈдёҖдёӢд»Җд№ҲжҳҜ Fragment д»ҘеҸҠе®ғжүҖи§ЈеҶізҡ„й—®йўҳгҖӮ

Fragment зҡ„еҜ“ж„ҸжҳҜиҰҒжёІжҹ“дёҖдёӘзүҮж®өпјҢеҒҮи®ҫжҲ‘д»¬жңүеҰӮдёӢжЁЎжқҝпјҡ

<template>
  <table>
    <tr>
      <Columns />
    </tr>
  </table>

еӨ§е®¶жҖқиҖғдёҖдёӘй—®йўҳпјҢеҰӮдёҠжЁЎжқҝзҡ„ VNode еҰӮдҪ•иЎЁзӨәпјҹеҰӮжһңжЁЎжқҝдёӯеҸӘжңүдёҖдёӘ td ж ҮзӯҫпјҢеҚіеҸӘжңүдёҖдёӘж №е…ғзҙ пјҢиҝҷеҫҲе®№жҳ“иЎЁзӨәпјҡ

    const elementVNode = {
      tag: 'td',
      data: null
    }

дҪҶжҳҜжЁЎжқҝдёӯдёҚд»…д»…еҸӘжңүдёҖдёӘ td ж ҮзӯҫпјҢиҖҢжҳҜжңүеӨҡдёӘ td ж ҮзӯҫпјҢеҚіеӨҡдёӘж №е…ғзҙ пјҢиҝҷеҰӮдҪ•иЎЁзӨәпјҹжӯӨж—¶жҲ‘д»¬е°ұйңҖиҰҒеј•е…ҘдёҖдёӘжҠҪиұЎе…ғзҙ пјҢд№ҹе°ұжҳҜжҲ‘д»¬иҰҒд»Ӣз»Қзҡ„ FragmentгҖӮ

const fragmentVNode = {

еҪ“жёІжҹ“еҷЁеңЁжёІжҹ“ VNode ж—¶пјҢеҰӮжһңеҸ‘зҺ°иҜҘ VNode зҡ„зұ»еһӢжҳҜ FragmentпјҢе°ұеҸӘйңҖиҰҒжҠҠиҜҘ VNode зҡ„еӯҗиҠӮзӮ№жёІжҹ“еҲ°йЎөйқўгҖӮ

TIP

еңЁдёҠйқўзҡ„д»Јз Ғдёӯ fragmentVNode.tag еұһжҖ§зҡ„еҖјжҳҜдёҖдёӘйҖҡиҝҮ Symbol еҲӣе»әзҡ„е”ҜдёҖж ҮиҜҶпјҢдҪҶе®һйҷ…дёҠжҲ‘д»¬жӣҙеҖҫеҗ‘дәҺз»ҷ VNode еҜ№иұЎж·»еҠ дёҖдёӘ flags еұһжҖ§пјҢз”ЁжқҘд»ЈиЎЁиҜҘ VNode зҡ„зұ»еһӢпјҢиҝҷеңЁжң¬з« зҡ„еҗҺйқўдјҡиҜҰз»ҶиҜҙжҳҺгҖӮ

еҶҚжқҘзңӢзңӢ PortalпјҢд»Җд№ҲжҳҜ Portal е‘ўпјҹ

иЎҘе……(зҺ°д»ЈеҒҡжі•): Vue 3 жӯЈејҸзүҲе·Іе°Ҷ Portal жӣҙеҗҚдёәеҶ…зҪ®з»„д»¶ <Teleport>пјҢз”Ёжі•дёә <Teleport to="#app-root">...</Teleport>пјҢж— йңҖжүӢеҠЁз»ҙжҠӨ Symbol ж ҮиҜҶпјҢжёІжҹ“еҷЁеҺҹз”ҹж”ҜжҢҒгҖӮ

дёҖеҸҘиҜқпјҡе®ғе…Ғи®ёдҪ жҠҠеҶ…е®№жёІжҹ“еҲ°д»»дҪ•ең°ж–№гҖӮе…¶еә”з”ЁеңәжҷҜжҳҜпјҢеҒҮи®ҫдҪ иҰҒе®һзҺ°дёҖдёӘи’ҷеұӮз»„д»¶ <Overlay/>пјҢиҰҒжұӮжҳҜиҜҘз»„д»¶зҡ„ z-index зҡ„еұӮзә§жңҖй«ҳпјҢиҝҷж ·ж— и®әеңЁе“ӘйҮҢдҪҝз”ЁйғҪеёҢжңӣе®ғиғҪеӨҹйҒ®дҪҸе…ЁйғЁеҶ…е®№пјҢдҪ еҸҜиғҪдјҡе°Ҷе…¶з”ЁеңЁд»»дҪ•дҪ йңҖиҰҒи’ҷеұӮзҡ„ең°ж–№гҖӮ

    <template>
      <div id="box" style="z-index: -1;">
        <Overlay />
      </div>
    </template>

еҰӮдёҠпјҢдёҚе№ёзҡ„дәӢжғ…еҸ‘з”ҹдәҶпјҢеңЁжІЎжңү Portal зҡ„жғ…еҶөдёӢпјҢдёҠйқўзҡ„ <Overlay/> з»„д»¶зҡ„еҶ…е®№еҸӘиғҪжёІжҹ“еҲ° id="box" зҡ„ div ж ҮзӯҫдёӢпјҢиҝҷе°ұдјҡеҜјиҮҙи’ҷеұӮзҡ„еұӮзә§еӨұж•Ҳз”ҡиҮіеёғеұҖйғҪеҸҜиғҪдјҡеҸ—еҲ°еҪұе“ҚгҖӮ

е…¶е®һи§ЈеҶіеҠһжі•д№ҹеҫҲз®ҖеҚ•пјҢеҒҮеҰӮ <Overlay/> з»„д»¶иҰҒжёІжҹ“зҡ„еҶ…е®№дёҚеҸ— DOM еұӮзә§е…ізі»йҷҗеҲ¶пјҢеҚіеҸҜд»ҘжёІжҹ“еҲ°д»»дҪ•дҪҚзҪ®пјҢиҜҘй—®йўҳе°ҶиҝҺеҲғиҖҢи§ЈгҖӮ

дҪҝз”Ё Portal еҸҜд»Ҙиҝҷж ·зј–еҶҷ <Overlay/> з»„д»¶зҡ„жЁЎжқҝпјҡ

<template>
  <Portal target="#app-root">
    <div class="overlay"></div>
  </Portal>
</template>

е…¶жңҖз»Ҳж•ҲжһңжҳҜпјҢж— и®әдҪ еңЁдҪ•еӨ„дҪҝз”Ё <Overlay/> з»„д»¶пјҢе®ғйғҪдјҡжҠҠеҶ…е®№жёІжҹ“еҲ° id="app-root" зҡ„е…ғзҙ дёӢгҖӮз”ұжӯӨеҸҜзҹҘпјҢжүҖи°“ Portal е°ұжҳҜжҠҠеӯҗиҠӮзӮ№жёІжҹ“еҲ°з»ҷе®ҡзҡ„зӣ®ж ҮпјҢжҲ‘д»¬еҸҜд»ҘдҪҝз”ЁеҰӮдёӢ VNode еҜ№иұЎжқҘжҸҸиҝ°дёҠйқўиҝҷж®өжЁЎжқҝпјҡ

const Portal = Symbol()
const portalVNode = {
  tag: Portal,
  data: {
    target: '#app-root'
  },
  children: {
    tag: 'div',
    data: {
      class: 'overlay'
    }
  }
}

Portal зұ»еһӢзҡ„ VNode дёҺ Fragment зұ»еһӢзҡ„ VNode зұ»дјјпјҢйғҪйңҖиҰҒдёҖдёӘе”ҜдёҖзҡ„ж ҮиҜҶпјҢжқҘеҢәеҲҶе…¶зұ»еһӢпјҢзӣ®зҡ„жҳҜе‘ҠиҜүжёІжҹ“еҷЁеҰӮдҪ•жёІжҹ“иҜҘ VNodeгҖӮ

VNode зҡ„з§Қзұ»

еҪ“ VNode жҸҸиҝ°дёҚеҗҢзҡ„дәӢзү©ж—¶пјҢе…¶еұһжҖ§зҡ„еҖјд№ҹеҗ„дёҚзӣёеҗҢгҖӮжҜ”еҰӮдёҖдёӘ VNode еҜ№иұЎжҳҜ html ж Үзӯҫзҡ„жҸҸиҝ°пјҢйӮЈд№Ҳе…¶ tag еұһжҖ§еҖје°ұжҳҜдёҖдёӘеӯ—з¬ҰдёІпјҢеҚіж Үзӯҫзҡ„еҗҚеӯ—пјӣеҰӮжһңжҳҜз»„д»¶зҡ„жҸҸиҝ°пјҢйӮЈд№Ҳе…¶ tag еұһжҖ§еҖјеҲҷеј•з”Ёз»„д»¶зұ»(жҲ–еҮҪж•°)жң¬иә«пјӣеҰӮжһңжҳҜж–Үжң¬иҠӮзӮ№зҡ„жҸҸиҝ°пјҢйӮЈд№Ҳе…¶ tag еұһжҖ§еҖјдёә nullгҖӮ

жңҖз»ҲжҲ‘д»¬еҸ‘зҺ°пјҢдёҚеҗҢзұ»еһӢзҡ„VNode жӢҘжңүдёҚеҗҢзҡ„и®ҫи®ЎпјҢиҝҷдәӣе·®ејӮз§Ҝе°‘жҲҗеӨҡпјҢжүҖд»ҘжҲ‘д»¬е®Ңе…ЁеҸҜд»Ҙе°Ҷе®ғд»¬еҲҶй—ЁеҲ«зұ»гҖӮ

жҖ»зҡ„жқҘиҜҙпјҢжҲ‘д»¬еҸҜд»ҘжҠҠ VNode еҲҶжҲҗдә”зұ»пјҢеҲҶеҲ«жҳҜпјҡhtml/svg е…ғзҙ гҖҒз»„д»¶ гҖҒзәҜж–Үжң¬ гҖҒFragment д»ҘеҸҠ Portal пјҡ

еҰӮдёҠеӣҫжүҖзӨәпјҢжҲ‘д»¬еҸҜд»ҘжҠҠз»„д»¶з»ҶеҲҶдёә жңүзҠ¶жҖҒз»„д»¶ е’Ң еҮҪж•°ејҸз»„д»¶ гҖӮеҗҢж—¶жңүзҠ¶жҖҒз»„д»¶иҝҳеҸҜд»Ҙз»ҶеҲҶдёәдёүйғЁеҲҶпјҡжҷ®йҖҡзҡ„жңүзҠ¶жҖҒз»„д»¶ гҖҒйңҖиҰҒиў« keepAlive зҡ„жңүзҠ¶жҖҒз»„д»¶ д»ҘеҸҠ е·Із»Ҹиў« keepAlive зҡ„жңүзҠ¶жҖҒз»„д»¶ гҖӮ

дҪҶж— и®әжҳҜжҷ®йҖҡзҡ„жңүзҠ¶жҖҒз»„д»¶иҝҳжҳҜ keepAlive зӣёе…ізҡ„жңүзҠ¶жҖҒз»„д»¶пјҢе®ғд»¬йғҪжҳҜжңүзҠ¶жҖҒз»„д»¶ гҖӮжүҖд»ҘжҲ‘д»¬еңЁи®ҫи®Ў VNode ж—¶еҸҜд»Ҙе°Ҷе®ғд»¬дҪңдёәдёҖзұ»зңӢеҫ…гҖӮ

дҪҝз”Ё flags дҪңдёә VNode зҡ„ж ҮиҜҶ

ж—ўз„¶ VNode жңүзұ»еҲ«д№ӢеҲҶпјҢжҲ‘д»¬е°ұжңүеҝ…иҰҒдҪҝз”ЁдёҖдёӘе”ҜдёҖзҡ„ж ҮиҜҶпјҢжқҘж ҮжҳҺжҹҗдёҖдёӘ VNode еұһдәҺе“ӘдёҖзұ»гҖӮеҗҢж—¶з»ҷ VNode ж·»еҠ  flags д№ҹжҳҜ Virtual DOM з®—жі•зҡ„дјҳеҢ–жүӢж®өд№ӢдёҖгҖӮ

жҜ”еҰӮеңЁ Vue2 дёӯеҢәеҲҶ VNode жҳҜ html е…ғзҙ иҝҳжҳҜз»„д»¶дәҰжҲ–жҳҜжҷ®йҖҡж–Үжң¬пјҢжҳҜиҝҷж ·еҒҡзҡ„пјҡ

  • 1гҖҒжӢҝеҲ° VNode еҗҺе…Ҳе°қиҜ•жҠҠе®ғеҪ“дҪңз»„д»¶еҺ»еӨ„зҗҶпјҢеҰӮжһңжҲҗеҠҹең°еҲӣе»әдәҶз»„д»¶пјҢйӮЈиҜҙжҳҺиҜҘ VNode е°ұжҳҜз»„д»¶зҡ„ VNode
  • 2гҖҒеҰӮжһңжІЎиғҪжҲҗеҠҹең°еҲӣе»әз»„д»¶пјҢеҲҷжЈҖжҹҘ vnode.tag жҳҜеҗҰжңүе®ҡд№үпјҢеҰӮжһңжңүе®ҡд№үеҲҷеҪ“дҪңжҷ®йҖҡж ҮзӯҫеӨ„зҗҶ
  • 3гҖҒеҰӮжһң vnode.tag жІЎжңүе®ҡд№үеҲҷжЈҖжҹҘжҳҜеҗҰжҳҜжіЁйҮҠиҠӮзӮ№
  • 4гҖҒеҰӮжһңдёҚжҳҜжіЁйҮҠиҠӮзӮ№пјҢеҲҷдјҡжҠҠе®ғеҪ“дҪңж–Үжң¬иҠӮзӮ№еҜ№еҫ…

д»ҘдёҠиҝҷдәӣеҲӨж–ӯйғҪжҳҜеңЁжҢӮиҪҪ(жҲ–patch)йҳ¶ж®өиҝӣиЎҢзҡ„пјҢжҚўеҸҘиҜқиҜҙпјҢдёҖдёӘ VNode еҲ°еә•жҸҸиҝ°зҡ„жҳҜд»Җд№ҲжҳҜеңЁжҢӮиҪҪжҲ– patch зҡ„ж—¶еҖҷжүҚзҹҘйҒ“зҡ„гҖӮиҝҷе°ұеёҰжқҘдәҶдёӨдёӘйҡҫйўҳпјҡж— жі•д»ҺAOT зҡ„еұӮйқўдјҳеҢ–гҖҒејҖеҸ‘иҖ…ж— жі•жүӢеҠЁдјҳеҢ– гҖӮ

дёәдәҶи§ЈеҶіиҝҷдёӘй—®йўҳпјҢжҲ‘д»¬зҡ„жҖқи·ҜжҳҜеңЁ VNode еҲӣе»әзҡ„ж—¶еҖҷе°ұжҠҠиҜҘ VNode зҡ„зұ»еһӢйҖҡиҝҮ flags ж ҮжҳҺпјҢиҝҷж ·еңЁжҢӮиҪҪжҲ– patch йҳ¶ж®өйҖҡиҝҮ flags еҸҜд»ҘзӣҙжҺҘйҒҝе…ҚжҺүеҫҲеӨҡж¶ҲиҖ—жҖ§иғҪзҡ„еҲӨж–ӯпјҢжҲ‘д»¬е…ҲжҸҗеүҚж„ҹеҸ—дёҖдёӢжёІжҹ“еҷЁзҡ„д»Јз Ғпјҡ

    if (flags & VNodeFlags.ELEMENT) {
      // VNode жҳҜжҷ®йҖҡж Үзӯҫ
      mountElement(/* ... */)
    } else if (flags & VNodeFlags.COMPONENT) {
      // VNode жҳҜз»„д»¶
      mountComponent(/* ... */)
    } else if (flags & VNodeFlags.TEXT) {
      // VNode жҳҜзәҜж–Үжң¬
      mountText(/* ... */)
    }

еҰӮдёҠпјҢйҮҮз”ЁдәҶдҪҚиҝҗз®—пјҢеңЁдёҖж¬ЎжҢӮиҪҪд»»еҠЎдёӯеҰӮдёҠеҲӨж–ӯеҫҲеҸҜиғҪеӨ§йҮҸзҡ„иҝӣиЎҢпјҢдҪҝз”ЁдҪҚиҝҗз®—еңЁдёҖе®ҡзЁӢеәҰдёҠеҶҚж¬ЎжӢүеҚҮдәҶиҝҗиЎҢж—¶жҖ§иғҪгҖӮ

TIP

е®һйҷ…дёҠ Vue3 еңЁ Virtual DOM зҡ„дјҳеҢ–дёҠйҮҮз”Ёзҡ„е°ұжҳҜ inferno (opens new window) зҡ„жүӢж®өгҖӮе…·дҪ“еҰӮдҪ•еҒҡжҲ‘д»¬дјҡеңЁеҗҺйқўзҡ„з« иҠӮд»Ӣз»ҚгҖӮ

иҝҷе°ұж„Ҹе‘ізқҖжҲ‘д»¬еңЁи®ҫи®Ў VNode еҜ№иұЎж—¶пјҢеә”иҜҘеҢ…еҗ« flags еӯ—ж®өпјҡ

    // VNode еҜ№иұЎ
    {
      flags: ...
    }

жһҡдёҫеҖј VNodeFlags

йӮЈд№ҲдёҖдёӘ VNode еҜ№иұЎзҡ„ flags еҸҜд»ҘжҳҜе“ӘдәӣеҖје‘ўпјҹйӮЈе°ұзңӢ VNode жңүе“Әдәӣз§Қзұ»е°ұеҘҪдәҶпјҢжҜҸдёҖдёӘ VNode з§Қзұ»жҲ‘д»¬йғҪдёәе…¶еҲҶй…ҚдёҖдёӘ flags еҖјеҚіеҸҜпјҢжҲ‘д»¬жҠҠе®ғи®ҫи®ЎжҲҗдёҖдёӘжһҡдёҫеҖје№¶еҸ–еҗҚдёә VNodeFlagsпјҢеңЁ javascript йҮҢе°ұз”ЁдёҖдёӘеҜ№иұЎжқҘиЎЁзӨәеҚіеҸҜпјҡ

    const VNodeFlags = {
      // html ж Үзӯҫ
      ELEMENT_HTML: 1,
      // SVG ж Үзӯҫ
      ELEMENT_SVG: 1 << 1,
    
      // жҷ®йҖҡжңүзҠ¶жҖҒз»„д»¶
      COMPONENT_STATEFUL_NORMAL: 1 << 2,
      // йңҖиҰҒиў«keepAliveзҡ„жңүзҠ¶жҖҒз»„д»¶
      COMPONENT_STATEFUL_SHOULD_KEEP_ALIVE: 1 << 3,
      // е·Із»Ҹиў«keepAliveзҡ„жңүзҠ¶жҖҒз»„д»¶
      COMPONENT_STATEFUL_KEPT_ALIVE: 1 << 4,
      // еҮҪж•°ејҸз»„д»¶
      COMPONENT_FUNCTIONAL: 1 << 5,
    
      // зәҜж–Үжң¬
      TEXT: 1 << 6,
      // Fragment
      FRAGMENT: 1 << 7,
      // Portal
      PORTAL: 1 << 8
    }

еҰӮдёҠиҝҷдәӣжһҡдёҫеұһжҖ§жүҖд»ЈиЎЁзҡ„ж„Ҹд№үиғҪеӨҹдёҺдёӢйқўзҡ„еӣҫзүҮдёҖдёҖеҜ№еә”дёҠпјҡ

жҲ‘д»¬жіЁж„ҸеҲ°пјҢиҝҷдәӣжһҡдёҫеұһжҖ§зҡ„еҖјеҹәжң¬йғҪжҳҜйҖҡиҝҮе°ҶеҚҒиҝӣеҲ¶ж•°еӯ— 1 е·Ұз§»дёҚеҗҢзҡ„дҪҚж•°еҫ—жқҘзҡ„гҖӮж №жҚ®иҝҷдәӣеҹәжң¬зҡ„жһҡдёҫеұһжҖ§еҖјпјҢжҲ‘д»¬иҝҳеҸҜд»Ҙжҙҫз”ҹеҮәйўқеӨ–зҡ„дёүдёӘж ҮиҜҶпјҡ

    // html е’Ң svg йғҪжҳҜж Үзӯҫе…ғзҙ пјҢеҸҜд»Ҙз”Ё ELEMENT иЎЁзӨә
    VNodeFlags.ELEMENT = VNodeFlags.ELEMENT_HTML | VNodeFlags.ELEMENT_SVG
    // жҷ®йҖҡжңүзҠ¶жҖҒз»„д»¶гҖҒйңҖиҰҒиў«keepAliveзҡ„жңүзҠ¶жҖҒз»„д»¶гҖҒе·Із»Ҹиў«keepAliceзҡ„жңүзҠ¶жҖҒз»„д»¶ йғҪжҳҜвҖңжңүзҠ¶жҖҒз»„д»¶вҖқпјҢз»ҹдёҖз”Ё COMPONENT_STATEFUL иЎЁзӨә
    VNodeFlags.COMPONENT_STATEFUL =
      VNodeFlags.COMPONENT_STATEFUL_NORMAL |
      VNodeFlags.COMPONENT_STATEFUL_SHOULD_KEEP_ALIVE |
      VNodeFlags.COMPONENT_STATEFUL_KEPT_ALIVE
    // жңүзҠ¶жҖҒз»„д»¶ е’Ң  еҮҪж•°ејҸз»„д»¶йғҪжҳҜвҖңз»„д»¶вҖқпјҢз”Ё COMPONENT иЎЁзӨә
    VNodeFlags.COMPONENT = VNodeFlags.COMPONENT_STATEFUL | VNodeFlags.COMPONENT_FUNCTIONAL

е…¶дёӯ VNodeFlags.ELEMENTгҖҒVNodeFlags.COMPONENT_STATEFUL д»ҘеҸҠ VNodeFlags.COMPONENT жҳҜз”ұеҹәжң¬ж ҮиҜҶйҖҡиҝҮжҢүдҪҚжҲ–(|)иҝҗз®—еҫ—еҲ°зҡ„пјҢиҝҷдёүдёӘжҙҫз”ҹеҖје°Ҷз”ЁдәҺиҫ…еҠ©еҲӨж–ӯгҖӮ

жңүдәҶиҝҷдәӣ flags д№ӢеҗҺпјҢжҲ‘д»¬еңЁеҲӣе»ә VNode зҡ„ж—¶еҖҷе°ұеҸҜд»Ҙйў„е…Ҳдёәе…¶жү“дёҠ flagsпјҢд»Ҙж ҮжҳҺиҜҘ VNode зҡ„зұ»еһӢпјҡ

    // html е…ғзҙ иҠӮзӮ№
    const htmlVnode = {
      flags: VNodeFlags.ELEMENT_HTML,
      tag: 'div',
      data: null
    }
    
    // svg е…ғзҙ иҠӮзӮ№
    const svgVnode = {
      flags: VNodeFlags.ELEMENT_SVG,
      tag: 'svg',
      data: null
    }
    
    // еҮҪж•°ејҸз»„д»¶
    const functionalComponentVnode = {
      flags: VNodeFlags.COMPONENT_FUNCTIONAL,
      tag: MyFunctionalComponent
    }
    
    // жҷ®йҖҡзҡ„жңүзҠ¶жҖҒз»„д»¶
    const normalComponentVnode = {
      flags: VNodeFlags.COMPONENT_STATEFUL_NORMAL,
      tag: MyStatefulComponent
    }
    
    // Fragment
    const fragmentVnode = {
      flags: VNodeFlags.FRAGMENT,
      // жіЁж„ҸпјҢз”ұдәҺ flags зҡ„еӯҳеңЁпјҢжҲ‘д»¬е·Із»ҸдёҚйңҖиҰҒдҪҝз”Ё tag еұһжҖ§жқҘеӯҳеӮЁе”ҜдёҖж ҮиҜҶ
      tag: null
    }
    
    // Portal
    const portalVnode = {
      flags: VNodeFlags.PORTAL,
      // жіЁж„ҸпјҢз”ұдәҺ flags зҡ„еӯҳеңЁпјҢжҲ‘д»¬е·Із»ҸдёҚйңҖиҰҒдҪҝз”Ё tag еұһжҖ§жқҘеӯҳеӮЁе”ҜдёҖж ҮиҜҶпјҢtag еұһжҖ§з”ЁжқҘеӯҳеӮЁ Portal зҡ„ target
      tag: target
    }

еҰӮдёӢжҳҜеҲ©з”Ё VNodeFlags еҲӨж–ӯ VNode зұ»еһӢзҡ„дҫӢеӯҗпјҢжҜ”еҰӮеҲӨж–ӯдёҖдёӘ VNode жҳҜеҗҰжҳҜз»„д»¶пјҡ

    // дҪҝз”ЁжҢүдҪҚдёҺ(&)иҝҗз®—
    functionalComponentVnode.flags & VNodeFlags.COMPONENT // зңҹ
    normalComponentVnode.flags & VNodeFlags.COMPONENT // зңҹ
    htmlVnode.flags & VNodeFlags.COMPONENT // еҒҮ

зҶҹжӮүдҪҚиҝҗз®—зҡ„иҜқпјҢзҗҶи§Јиө·жқҘеҫҲз®ҖеҚ•гҖӮиҝҷе®һйҷ…дёҠжҳҜеӨҡз§ҚдҪҚиҝҗз®—жҠҖе·§дёӯзҡ„дёҖдёӘе°ҸжҠҖе·§гҖӮжҲ‘д»¬еҸҜд»ҘеҲ—дёҖдёӘиЎЁж јпјҡ

VNodeFlagsе·Ұз§»иҝҗз®—32 дҪҚзҡ„ bit еәҸеҲ—(еҮәдәҺз®Җз•ҘпјҢеҸӘз”Ё 9 дҪҚиЎЁзӨә)
ELEMENT_HTMLж— 000000001
ELEMENT_SVG1 << 1000000010
COMPONENT_STATEFUL_NORMAL1 << 2000000100
COMPONENT_STATEFUL_SHOULD_KEEP_ALIVE1 << 3000001000
COMPONENT_STATEFUL_KEPT_ALIVE1 << 4000010000
COMPONENT_FUNCTIONAL1 << 5000100000
TEXT1 << 6001000000
FRAGMENT1 << 7010000000
PORTAL1 << 8100000000

ж №жҚ®дёҠиЎЁеұ•зӨәзҡ„еҹәжң¬ flags еҖјеҸҜд»ҘеҫҲе®№жҳ“ең°еҫ—еҮәдёӢиЎЁпјҡ

VNodeFlags32 дҪҚзҡ„жҜ”зү№еәҸеҲ—(еҮәдәҺз®Җз•ҘпјҢеҸӘз”Ё 9 дҪҚиЎЁзӨә)
ELEMENT00000001 1
COMPONENT_STATEFUL00001 1 100
COMPONENT0001 1 1 100

жүҖд»ҘеҫҲиҮӘз„¶зҡ„пјҢеҸӘжңү VNodeFlags.ELEMENT_HTML е’Ң VNodeFlags.ELEMENT_SVG дёҺ VNodeFlags.ELEMENT иҝӣиЎҢжҢүдҪҚдёҺ(&)иҝҗз®—жүҚдјҡеҫ—еҲ°йқһйӣ¶еҖјпјҢеҚідёәзңҹгҖӮ

children е’Ң ChildrenFlags

DOM жҳҜдёҖжЈөж ‘ж—©е·Іе®¶иҮідәәиҜҙпјҢж—ўз„¶ VNode жҳҜзңҹе®һжёІжҹ“еҶ…е®№зҡ„жҸҸиҝ°пјҢйӮЈд№Ҳе®ғеҝ…з„¶д№ҹжҳҜдёҖжЈөж ‘гҖӮеңЁд№ӢеүҚзҡ„и®ҫи®ЎдёӯпјҢжҲ‘д»¬з»ҷ VNode е®ҡд№үдәҶ children еұһжҖ§пјҢз”ЁжқҘеӯҳеӮЁеӯҗ VNodeгҖӮеӨ§е®¶жҖқиҖғдёҖдёӢпјҢдёҖдёӘж Үзӯҫзҡ„еӯҗиҠӮзӮ№дјҡжңүеҮ з§Қжғ…еҶөпјҹ

жҖ»зҡ„жқҘиҜҙж— йқһжңүд»ҘдёӢеҮ з§Қпјҡ

  • жІЎжңүеӯҗиҠӮзӮ№
  • еҸӘжңүдёҖдёӘеӯҗиҠӮзӮ№
  • еӨҡдёӘеӯҗиҠӮзӮ№
    • жңү key
    • ж—  key
  • дёҚзҹҘйҒ“еӯҗиҠӮзӮ№зҡ„жғ…еҶө

жҲ‘д»¬еҸҜд»Ҙз”ЁдёҖдёӘеҸ«еҒҡ ChildrenFlags зҡ„еҜ№иұЎжқҘжһҡдёҫеҮәд»ҘдёҠиҝҷдәӣжғ…еҶөпјҢдҪңдёәдёҖдёӘ VNode зҡ„еӯҗиҠӮзӮ№зҡ„зұ»еһӢж ҮиҜҶпјҡ

    const ChildrenFlags = {
      // жңӘзҹҘзҡ„ children зұ»еһӢ
      UNKNOWN_CHILDREN: 0,
      // жІЎжңү children
      NO_CHILDREN: 1,
      // children жҳҜеҚ•дёӘ VNode
      SINGLE_VNODE: 1 << 1,
    
      // children жҳҜеӨҡдёӘжӢҘжңү key зҡ„ VNode
      KEYED_VNODES: 1 << 2,
      // children жҳҜеӨҡдёӘжІЎжңү key зҡ„ VNode
      NONE_KEYED_VNODES: 1 << 3
    }

з”ұдәҺ ChildrenFlags.KEYED_VNODES е’Ң ChildrenFlags.NONE_KEYED_VNODES йғҪеұһдәҺеӨҡдёӘ VNodeпјҢжүҖд»ҘжҲ‘д»¬еҸҜд»Ҙжҙҫз”ҹеҮәдёҖдёӘвҖңеӨҡиҠӮзӮ№вҖқж ҮиҜҶпјҢд»Ҙж–№дҫҝзЁӢеәҸзҡ„еҲӨж–ӯпјҡ

    ChildrenFlags.MULTIPLE_VNODES = ChildrenFlags.KEYED_VNODES | ChildrenFlags.NONE_KEYED_VNODES

иҝҷж ·жҲ‘д»¬еҲӨж–ӯдёҖдёӘ VNode зҡ„еӯҗиҠӮзӮ№жҳҜеҗҰжҳҜеӨҡдёӘеӯҗиҠӮзӮ№е°ұеҸҳеҫ—е®№жҳ“еӨҡдәҶпјҡ

    someVNode.childFlags & ChildrenFlags.MULTIPLE_VNODES

TIP

дёәд»Җд№Ҳ children д№ҹйңҖиҰҒж ҮиҜҶе‘ўпјҹеҺҹеӣ еҸӘжңүдёҖдёӘпјҡдёәдәҶдјҳеҢ– гҖӮеңЁеҗҺйқўи®Іи§Ј diff з®—жі•зҡ„з« иҠӮдёӯдҪ е°Ҷдјҡж„ҸиҜҶеҲ°пјҢиҝҷдәӣдҝЎжҒҜжҳҜиҮіе…ійҮҚиҰҒзҡ„гҖӮ

еңЁдёҖдёӘ VNode еҜ№иұЎдёӯпјҢжҲ‘д»¬дҪҝз”Ё flags еұһжҖ§жқҘеӯҳеӮЁиҜҘ VNode зҡ„зұ»еһӢпјҢзұ»дјјзҡ„пјҢжҲ‘д»¬е°ҶдҪҝз”Ё childFlags жқҘеӯҳеӮЁеӯҗиҠӮзӮ№зҡ„зұ»еһӢпјҢжҲ‘д»¬жқҘдёҫдёҖдәӣе®һйҷ…зҡ„дҫӢеӯҗпјҡ

    // жІЎжңүеӯҗиҠӮзӮ№зҡ„ div ж Үзӯҫ
    const elementVNode = {
      flags: VNodeFlags.ELEMENT_HTML,
      tag: 'div',
      data: null,
      children: null,
      childFlags: ChildrenFlags.NO_CHILDREN
    }
    
    // ж–Үжң¬иҠӮзӮ№зҡ„ childFlags е§Ӣз»ҲйғҪжҳҜ NO_CHILDREN
    const textVNode = {
      tag: null,
      data: null,
      children: 'жҲ‘жҳҜж–Үжң¬',
      childFlags: ChildrenFlags.NO_CHILDREN
    }
    
    // жӢҘжңүеӨҡдёӘдҪҝз”ЁдәҶkeyзҡ„ li ж ҮзӯҫдҪңдёәеӯҗиҠӮзӮ№зҡ„ ul ж Үзӯҫ
    const elementVNode = {
      flags: VNodeFlags.ELEMENT_HTML,
      tag: 'ul',
      data: null,
      childFlags: ChildrenFlags.KEYED_VNODES,
      children: [
        {
          tag: 'li',
          data: null,
          key: 0
        },
        {
          tag: 'li',
          data: null,
          key: 1
        }
      ]
    }
    
    // еҸӘжңүдёҖдёӘеӯҗиҠӮзӮ№зҡ„ Fragment
    const elementVNode = {
      flags: VNodeFlags.FRAGMENT,
      tag: null,
      data: null,
      childFlags: ChildrenFlags.SINGLE_VNODE,
      children: {
        tag: 'p',
        data: null
      }
    }

дҪҶе№¶йқһжүҖжңүзұ»еһӢзҡ„ VNode зҡ„ children еұһжҖ§йғҪжҳҜз”ЁжқҘеӯҳеӮЁеӯҗ VNodeпјҢжҜ”еҰӮз»„д»¶зҡ„вҖңеӯҗ VNodeвҖқе…¶е®һдёҚеә”иҜҘдҪңдёә children иҖҢжҳҜеә”иҜҘдҪңдёә slotsпјҢжүҖд»ҘжҲ‘д»¬дјҡе®ҡд№ү VNode.slots еұһжҖ§жқҘеӯҳеӮЁиҝҷдәӣеӯҗ VNodeпјҢдёҚиҝҮзӣ®еүҚжқҘиҜҙжҲ‘д»¬иҝҳдёҚйңҖиҰҒж·ұе…ҘжҺўи®Ёжңүе…іжҸ’ж§Ҫзҡ„зҹҘиҜҶгҖӮ

VNodeData

еүҚйқўжҸҗеҲ°иҝҮпјҢVNodeData жҢҮзҡ„жҳҜ VNode зҡ„ data еұһжҖ§пјҢе®ғжҳҜдёҖдёӘеҜ№иұЎпјҡ

{
  flags: ...,
  tag: ...,
  // VNodeData
  data: {
    ...
  }
}

VNodeData йЎҫеҗҚжҖқд№үпјҢе®ғе°ұжҳҜ VNode ж•°жҚ®пјҢз”ЁдәҺеҜ№ VNode иҝӣиЎҢжҸҸиҝ°гҖӮдёҫдёӘдҫӢеӯҗпјҢеҒҮеҰӮдёҖдёӘ VNode зҡ„зұ»еһӢжҳҜ html ж ҮзӯҫпјҢеҲҷ VNodeData дёӯеҸҜд»ҘеҢ…еҗ« classгҖҒstyle д»ҘеҸҠдёҖдәӣдәӢд»¶пјҢиҝҷж ·жёІжҹ“еҷЁеңЁжёІжҹ“жӯӨ VNode ж—¶пјҢжүҚзҹҘйҒ“иҝҷдёӘж Үзӯҫзҡ„иғҢжҷҜйўңиүІгҖҒеӯ—дҪ“еӨ§е°Ҹд»ҘеҸҠзӣ‘еҗ¬дәҶе“ӘдәӣдәӢд»¶зӯүзӯүгҖӮжүҖд»Ҙд»Һи®ҫи®Ўи§’еәҰжқҘи®ІпјҢд»»дҪ•еҸҜд»ҘеҜ№ VNode иҝӣиЎҢжҸҸиҝ°зҡ„еҶ…е®№пјҢжҲ‘д»¬йғҪеҸҜд»Ҙе°Ҷе…¶еӯҳж”ҫеҲ° VNodeData еҜ№иұЎдёӯпјҢеҰӮпјҡ

{
  flags: VNodeFlags.ELEMENT_HTML,
  tag: 'div',
  data: {
    class: ['class-a', 'active'],
    style: {
      background: 'red',
      color: 'green'
    },
    // е…¶д»–ж•°жҚ®...
  }
}

еҰӮжһң VNode зҡ„зұ»еһӢжҳҜз»„д»¶пјҢйӮЈд№ҲжҲ‘д»¬еҗҢж ·еҸҜд»Ҙз”Ё VNodeData жқҘжҸҸиҝ°з»„д»¶пјҢжҜ”еҰӮз»„д»¶зҡ„дәӢд»¶гҖҒз»„д»¶зҡ„ props зӯүзӯүпјҢеҒҮи®ҫжңүеҰӮдёӢжЁЎжқҝпјҡ

    <MyComponent @some-event="handler" prop-a="1" />

еҲҷе…¶еҜ№еә”зҡ„ VNodeData еә”дёәпјҡ

{
  flags: VNodeFlags.COMPONENT_STATEFUL,
  tag: 'div',
  data: {
    on: {
      'some-event': handler
    },
    propA: '1'
    // е…¶д»–ж•°жҚ®...
  }
}

еҪ“з„¶дәҶпјҢеҸӘиҰҒиғҪеӨҹжӯЈзЎ®ең°еҜ№ VNode иҝӣиЎҢжҸҸиҝ°пјҢе…·дҪ“зҡ„ж•°жҚ®з»“жһ„дҪ еҸҜд»ҘйҡҸж„Ҹи®ҫи®ЎгҖӮжҲ‘д»¬жҡӮдё”дёҚйҷҗеҲ¶ VNodeData зҡ„еӣәе®ҡж јејҸгҖӮ

еңЁеҗҺз»ӯз« иҠӮдёӯпјҢжҲ‘д»¬дјҡж №жҚ®йңҖжұӮйҖҗжёҗең°е®Ңе–„ VNodeData зҡ„и®ҫи®ЎгҖӮ

иҮіжӯӨпјҢжҲ‘д»¬е·Із»ҸеҜ№ VNode е®ҢжҲҗдәҶдёҖе®ҡзҡ„и®ҫи®ЎпјҢзӣ®еүҚдёәжӯўжҲ‘д»¬жүҖи®ҫи®Ўзҡ„ VNode еҜ№иұЎеҰӮдёӢпјҡ

    export interface VNode {
      // _isVNode еұһжҖ§еңЁдёҠж–ҮдёӯжІЎжңүжҸҗеҲ°пјҢе®ғжҳҜдёҖдёӘе§Ӣз»Ҳдёә true зҡ„еҖјпјҢжңүдәҶе®ғпјҢжҲ‘д»¬е°ұеҸҜд»ҘеҲӨж–ӯдёҖдёӘеҜ№иұЎжҳҜеҗҰжҳҜ VNode еҜ№иұЎ
      _isVNode: true
      // el еұһжҖ§еңЁдёҠж–Үдёӯд№ҹжІЎжңүжҸҗеҲ°пјҢеҪ“дёҖдёӘ VNode иў«жёІжҹ“дёәзңҹе®һ DOM д№ӢеҗҺпјҢel еұһжҖ§зҡ„еҖјдјҡеј•з”ЁиҜҘзңҹе®һDOM
      el: Element | null
      flags: VNodeFlags
      tag: string | FunctionalComponent | ComponentClass | null
      data: VNodeData | null
      children: VNodeChildren
      childFlags: ChildrenFlags
    }

е…¶дёӯ _isVNode еұһжҖ§е’Ң el еұһжҖ§еңЁдёҠж–ҮдёӯжІЎжңүжҸҗеҲ°пјҢ_isVNode еұһжҖ§жҳҜдёҖдёӘе§Ӣз»Ҳдёә true зҡ„еҖјпјҢжңүдәҶе®ғпјҢжҲ‘д»¬е°ұеҸҜд»ҘеҲӨж–ӯдёҖдёӘеҜ№иұЎжҳҜеҗҰжҳҜ VNode еҜ№иұЎгҖӮel еұһжҖ§зҡ„еҖјеңЁ VNode иў«жёІжҹ“дёәзңҹе®һ DOM д№ӢеүҚдёҖзӣҙйғҪжҳҜ nullпјҢеҪ“ VNode иў«жёІжҹ“дёәзңҹе®һ DOM д№ӢеҗҺпјҢel еұһжҖ§зҡ„еҖјдјҡеј•з”ЁиҜҘзңҹе®һ DOMгҖӮ

е®һйҷ…дёҠпјҢеҰӮжһңдҪ зңӢиҝҮ Vue3 зҡ„жәҗз ҒпјҢдҪ дјҡеҸ‘зҺ°еңЁжәҗз ҒдёӯдёҖдёӘ VNode еҜ№иұЎйҷӨдәҶеҢ…еҗ«жң¬иҠӮжҲ‘д»¬жүҖи®ІеҲ°зҡ„иҝҷдәӣеұһжҖ§д№ӢеӨ–пјҢиҝҳеҢ…еҗ«иҜёеҰӮ handle е’Ң contextVNodeгҖҒparentVNodeгҖҒkeyгҖҒrefгҖҒslots зӯүе…¶д»–йўқеӨ–зҡ„еұһжҖ§гҖӮ

жҲ‘д»¬д№ӢжүҖд»ҘжІЎжңүеңЁжң¬з« дёӯеҢ…еҗ«иҝҷдәӣеҶ…е®№пјҢжҳҜеӣ дёәзӣ®еүҚжқҘи®ІпјҢжҲ‘д»¬ж №жң¬дёҚйңҖиҰҒиҝҷдәӣеұһжҖ§пјҢжҜ”еҰӮ handle еұһжҖ§д»…з”ЁдәҺеҮҪж•°ејҸз»„д»¶пјҢжүҖд»ҘжҲ‘д»¬дјҡеңЁеҮҪж•°ејҸз»„д»¶еҺҹзҗҶзӣёе…ізҡ„з« иҠӮеҶҚи®ІгҖӮ
